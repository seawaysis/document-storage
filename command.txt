// nest js
g = generate
mo = models
s = services
co = controller

nest g mo [module_name] //create module path => src/[module_name]

nest g s [module_name]/services/[module_name] --flat --no-spec //create service
nest g co [module_name]/controllers/[module_name] --flat --no-spec // create controller
